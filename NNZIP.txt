"""Core save/load logic for ZIP-based neural network checkpoints.""" import io, json, csv, pickle, struct, zipfile from datetime import datetime from pathlib import Path from typing import Any, Dict, List, Optional import numpy as np _ARCH_FILE = "architecture.json" _WEIGHTS_FILE = "weights.bin" _OPT_FILE = "optimizer.pkl" _META_FILE = "metadata.json" _LOG_FILE = "training_log.csv" _LOG_FIELDS = ["epoch","train_loss","val_loss","train_acc","val_acc","lr","timestamp"] def _encode_weights(state_dict: Dict[str, np.ndarray]) -> bytes: """ Serialise {name: ndarray} to compact binary. Layout per tensor: [name_len:u32][name:utf8][dtype_len:u8][dtype:ascii] [ndim:u8][shape:ndim*u64][data:raw bytes] """ buf = io.BytesIO() for name, arr in state_dict.items(): arr = np.asarray(arr) enc_name = name.encode("utf-8") enc_dtype = arr.dtype.str.encode("ascii") buf.write(struct.pack(">I", len(enc_name))) buf.write(enc_name) buf.write(struct.pack(">B", len(enc_dtype))) buf.write(enc_dtype) buf.write(struct.pack(">B", arr.ndim)) buf.write(struct.pack(f">{arr.ndim}Q", *arr.shape)) buf.write(arr.tobytes()) return buf.getvalue() def _decode_weights(data: bytes) -> Dict[str, np.ndarray]: buf, state = io.BytesIO(data), {} while True: header = buf.read(4) if not header: break name_len, = struct.unpack(">I", header) name = buf.read(name_len).decode("utf-8") dtype_len, = struct.unpack(">B", buf.read(1)) dtype = np.dtype(buf.read(dtype_len).decode("ascii")) ndim, = struct.unpack(">B", buf.read(1)) shape = struct.unpack(f">{ndim}Q", buf.read(8 * ndim)) nbytes = int(np.prod(shape)) * dtype.itemsize state[name] = np.frombuffer(buf.read(nbytes), dtype=dtype).reshape(shape) return state class Checkpoint: """ZIP-backed neural network checkpoint.""" def __init__(self, architecture, state_dict, optimizer_state=None, metadata=None, training_log=None): self.architecture = architecture self.state_dict = state_dict self.optimizer_state = optimizer_state zf.writestr(_WEIGHTS_FILE, _encode_weights(self.state_dict)) if self.optimizer_state is not None: zf.writestr(_OPT_FILE, pickle.dumps(self.optimizer_state)) meta = {**self.metadata, "_saved_at": datetime.utcnow().isoformat()} zf.writestr(_META_FILE, json.dumps(meta, indent=2)) if self.training_log: buf = io.StringIO() w = csv.DictWriter(buf, fieldnames=_LOG_FIELDS, extrasaction="ignore") w.writeheader(); w.writerows(self.training_log) zf.writestr(_LOG_FILE, buf.getvalue()) return path.resolve() @classmethod def load(cls, path: str) -> "Checkpoint": path = Path(path) if not path.exists(): raise FileNotFoundError(f"Checkpoint not found: {path}") with zipfile.ZipFile(path, "r") as zf: names = set(zf.namelist()) architecture = json.loads(zf.read(_ARCH_FILE)) state_dict = val_loss=float("nan"), train_acc=float("nan"), val_acc=float("nan"), lr=float("nan")): self.training_log.append({ "epoch": epoch, "train_loss": round(train_loss, 6), "val_loss": round(val_loss, 6), "train_acc": round(train_acc, 6), "val_acc": round(val_acc, 6), "lr": lr, "timestamp": datetime.utcnow().isoformat(), }) def best_epoch(self, metric="val_loss", mode="min"): if not self.training_log: return None fn = min if mode == "min" else max return fn(self.training_log, key=lambda r: float(r[metric])) def summary(self) -> str: lines = ["=== Checkpoint summary ===", f" Tensors : nn_checkpoint/manager.py

""" CheckpointManager — keeps the N best checkpoints on disk, deletes the rest, and supports resuming a training run. """ import json from pathlib import Path from typing import Optional from .checkpoint import Checkpoint class CheckpointManager: """ Manages a directory of checkpoint ZIPs. Parameters ---------- directory : str Folder to store checkpoint files in. keep_top_n : int Max number of checkpoints to retain (by val_loss, lower is better). metric : str Log field used to rank checkpoints. mode : str 'min' or 'max'. """ def __init__(self, directory: str, keep_top_n: int = 3, metric: str = "val_loss", mode: str = "min"): self.directory = Path(directory) self.directory.mkdir(parents=True, exist_ok=True) self.keep_top_n = keep_top_n self.metric = metric self.mode = mode self._index_path = self.directory / "index.json" self._index: list = self._load_index() def _load_index(self) -> list: if self._index_path.exists(): return json.loads(self._index_path.read_text()) return [] def _save_index(self): self._index_path.write_text(json.dumps(self._index, indent=2)) def save(self, ckpt: Checkpoint, epoch: int) -> Optional[Path]: """ Save *ckpt* if it is better than the current worst kept checkpoint. Returns the saved path, or None if the checkpoint was not kept. """ best = ckpt.best_epoch(self.metric, self.mode) if best is None: return None score = float(best[self.metric]) filename = f"epoch_{epoch:04d}.zip" path = self.directory / filename # Check if worth saving if len(self._index) >= self.keep_top_n: worst = self._worst() if self.mode == "min" and score >= worst["score"]: return None if self.mode == "max" and score <= worst["score"]: return None ckpt.save(path) self._index.append({"epoch": epoch, "score": score, "file": filename}) self._prune() self._save_index() return path def _worst(self) -> dict: fn = max if self.mode == "min" else min return fn(self._index, key=lambda r: r["score"]) def _prune(self): """Delete lowest-ranked checkpoints beyond keep_top_n.""" if len(self._index) <= self.keep_top_n: return reverse = (self.mode == "max") self._index.sort(key=lambda r: r["score"], reverse=reverse) to_remove = self._index[self.keep_top_n:] for entry in to_remove: p = self.directory / entry["file"] if p.exists(): p.unlink() self._index = self._index[:self.keep_top_n] def best(self) -> Optional[Checkpoint]: """Load and return the best checkpoint on disk.""" if not self._index: return None fn = min if self.mode == "min" else max entry = fn(self._index, key=lambda r: r["score"]) return Checkpoint.load(self.directory / entry["file"]) def latest_epoch(self) -> int: """Return the highest epoch number saved, or 0 if none.""" if not self._index: return 0 return max(e["epoch"] for e in self._index)

""" CLI tool: python -m nn_checkpoint.inspect path/to/checkpoint.zip """ import sys, zipfile, json from pathlib import Path from .checkpoint import Checkpoint def main(): if len(sys.argv) < 2: print("Usage: python -m nn_checkpoint.inspect <checkpoint.zip>") sys.exit(1) path = Path(sys.argv[1]) # Peek at raw zip contents first print(f"\nZIP contents of {path.name}:") with zipfile.ZipFile(path) as zf: for info in zf.infolist(): print(f" {info.filename:30s} {info.file_size:>10,} B " f"→ {info.compress_size:>10,} B compressed") print() ckpt = Checkpoint.load(path) print(ckpt.summary()) if ckpt.training_log: print("\nTraining log (last 5 epochs):") header = f"{'epoch':>6} {'train_loss':>12} {'val_loss':>10} {'val_acc':>8} {'lr':>10}" print(header) print("-" * len(header)) for row in ckpt.training_log[-5:]: print(f"{row['epoch']:>6} {row['train_loss']:>12} " f"{row['val_loss']:>10} {row['val_acc']:>8} {row['lr']:>10}") best = ckpt.best_epoch() if best: print(f"\nBest epoch: {best['epoch']} val_loss={best['val_loss']}") if __name__ == "__main__": main() 

examples/train_loop.py

""" End-to-end demo: simulate a training run and checkpoint it. """ import numpy as np from nn_checkpoint.checkpoint import Checkpoint from nn_checkpoint.manager import CheckpointManager # ── Fake network ───────────────────────────────────────────────────────────── architecture = { "layers": [ {"type": "Linear", "in": 784, "out": 256, "activation": "relu"}, {"type": "Linear", "in": 256, "out": 128, "activation": "relu"}, {"type": "Linear", "in": 128, "out": 10, "activation": "softmax"}, ] } def random_state_dict(): return { "fc1.weight": np.random.randn(256, 784).astype(np.float32), "fc1.bias": np.random.randn(256).astype(np.float32), "fc2.weight": np.random.randn(128, 256).astype(np.float32), "fc2.bias": np.random.randn(128).astype(np.float32), "fc3.weight": np.random.randn( 10, 128).astype(np.float32), "fc3.bias": np.random.randn( 10).astype(np.float32), } # ── Simulate training ───────────────────────────────────────────────────────── manager = CheckpointManager("checkpoints/", keep_top_n=3) ckpt = Checkpoint(architecture, random_state_dict(), metadata={"task": "mnist"}) for epoch in range(1, 11): # Simulated metrics train_loss = 2.0 * (0.75 ** epoch) + np.random.uniform(0, 0.05) val_loss = train_loss + np.random.uniform(0, 0.02) val_acc = 1.0 - val_loss / 2.0 lr = 1e-3 * (0.9 ** epoch) # Update weights (would come from your optimiser step) ckpt.state_dict = random_state_dict() ckpt.metadata["epoch"] = epoch # Log metrics ckpt.log_epoch(epoch, train_loss, val_loss=val_loss, val_acc=val_acc, lr=lr) # Checkpoint manager saves only the top-3 by val_loss saved = manager.save(ckpt, epoch) status = f"saved → {saved.name}" if saved else "not kept" print(f"Epoch {epoch:02d} val_loss={val_loss:.4f} {status}") # ── Resume / inspect ────────────────────────────────────────────────────────── print("\nLoading best checkpoint…") best = manager.best() print(best.summary()) print(f"\nBest epoch: {best.best_epoch()}")

Key design decisions

Custom binary format for weights — rather than pickling the whole state dict, each tensor is encoded with its name, dtype, shape, and raw bytes separately. This makes the binary chunk compress far better with DEFLATE (float32 near-zero weights have lots of repeated byte patterns), and means you can swap out individual tensors without re-reading the whole file.

CheckpointManager keeps only the top N — during a long training run you don't want hundreds of checkpoint ZIPs filling your disk. The manager ranks by your chosen metric and prunes automatically, maintaining an index.json so it can resume cleanly after a crash.

Everything is framework-agnostic — the only dependency is NumPy. Plugging in PyTorch is one line: state_dict = {k: v.cpu().numpy() for k, v in model.state_dict().items()}.
